Attack Counts By Region data frame

attack_count = excel_path |>
  group_by(region_txt, iyear) |>
  summarise(total_count=n(), .groups = 'drop') |>
  as.data.frame()

ggplot(attack_count, aes(x=iyear, y=total_count, group=region_txt, col=region_txt)) +
